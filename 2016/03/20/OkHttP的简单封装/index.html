OkHttP的简单封装

  • android
  • 开源
学习

最近项目要求把网络访问那一块改为使用OkHttp，于是就把Okhttp的基本使用封装了一下。关于okttp的异步请求、缓存以及拦截器等其他功能用法后续补充。

package com.belief.androidlearning.OkHttp;

import android.net.Uri;
import android.text.TextUtils;

import com.belief.androidlearning.util.FileCopyUtil;

import java.io.BufferedInputStream;
import java.io.BufferedReader;
import java.io.File;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.net.FileNameMap;
import java.net.URLConnection;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.concurrent.TimeUnit;
import java.util.zip.GZIPInputStream;

import okhttp3.Call;
import okhttp3.Cookie;
import okhttp3.CookieJar;
import okhttp3.FormBody;
import okhttp3.Headers;
import okhttp3.HttpUrl;
import okhttp3.MediaType;
import okhttp3.MultipartBody;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

/**
 * Created by wh on 2016/3/20.
 */
public class OkHttpUtil {
    public static String TAG = OkHttpUtil.class.getSimpleName();

    private static OkHttpUtil mInstance;
    private OkHttpClient mOkHttpClient;


    private OkHttpUtil(){
        mOkHttpClient = new OkHttpClient.Builder()
        .connectTimeout(30, TimeUnit.SECONDS)
        .writeTimeout(10,TimeUnit.SECONDS)
        .readTimeout(30, TimeUnit.SECONDS)
        //Cookie管理，Coolie持久化方案可以参考：https://github.com/franmontiel/PersistentCookieJar
        .cookieJar(new CookieJar() {
            private final HashMap<HttpUrl, List<Cookie>> cookieStore = new HashMap<>();
            @Override
            public void saveFromResponse(HttpUrl url, List<Cookie> cookies) {
                cookieStore.put(url, cookies);
            }

            @Override
            public List<Cookie> loadForRequest(HttpUrl url) {
                List<Cookie> cookies = cookieStore.get(url);
                return cookies != null ? cookies : new ArrayList<Cookie>();
            }
        })
        .build();
    }

    public static OkHttpUtil getInstance(){
        if(mInstance == null){
            synchronized (OkHttpUtil.class){
                if(mInstance == null){
                    mInstance = new OkHttpUtil();
                }
            }
        }
        return mInstance;
    }

    //同步Get请求
    public static String get(String url, List<Param> params) throws IOException{
        return get(url, params, false);
    }

    public static String get(String url, List<Param> params, boolean isAcceptGzipResponse) throws IOException{
        String result = "";
        Response response = null;
        try {
            response = getInstance().getRequest(url, params);
            if(response.isSuccessful()){
                if("gzip".equals(response.header("Content-Encoding")) || isAcceptGzipResponse){
                    InputStream inputStream = response.body().byteStream();
                    BufferedInputStream bis = new BufferedInputStream(inputStream);
                    inputStream = new GZIPInputStream(bis);
                    InputStreamReader inputStreamReader = new InputStreamReader(inputStream);
                    BufferedReader br = new BufferedReader(inputStreamReader);
                    StringBuffer sb = new StringBuffer("");
                    String line;
                    while ((line = br.readLine()) != null) {
                        sb.append(line);
                        sb.append("\n");
                    }
                    inputStreamReader.close();
                    br.close();
                    result = sb.toString();
                }else{
                    //没进行压缩
                    result = response.body().string();
                }
            }
        } catch (IOException e) {
            throw e;
        }
        return result;
    }

    /**
     * post请求，内部已经对参数urlencoding编码，不用再编码
     * @param url url最好是不带参数的基本的url,如果带有<code>?</code>,则至少要有一对参数
     * @param params get请求的参数都可以放在这里,null或者空List表示没有参数
     * @return String 默认返回空字符串
     */
    public static String post(String url, List<Param> params) throws IOException{
        String result = "";
        Response response = null;
        try {
            response = getInstance().postRequest(url, params);
            if(response.isSuccessful()){
                result = response.body().string();
            }
        } catch (IOException e) {
            throw e;
        }
        return result;
    }

    //post json数据
    public static String postString(String url, String entityString) throws IOException{
        String result = "";
        Response response = null;
        try {
            response = getInstance().postRequest(url, entityString);
            if(response.isSuccessful()){
                result = response.body().string();
            }
        } catch (IOException e) {
            throw e;
        }
        return result;
    }

    /**
     * 该方法是为了解决服务端没有返回码的情况，通过返回值来判断是否请求成功
     * @param url
     * @param entityString
     * @return
     */
    public static int postNullResultRequest(String url, String entityString){
        Response response = null;
        try {
            response = getInstance().postRequest(url, entityString);
        } catch (IOException e) {
        }
        return response.code();
    }

    /**
     * 向服务端提交参数为<code>xml</code>的POST请求
     * @param url 要提交数据的url
     * @param xmlValue 参数为xml对应的值
     * @return 返回数据，异常返回null
     */
    public static String postXmlRequest(String url,String xmlValue) throws IOException {
        List<Param> params = new ArrayList<>(1);
        Param param = new Param("xml", xmlValue);
        params.add(param);
        return post(url, params);
    }

    public static String uploadFile(String url, File file) throws IOException{
        return uploadFile(url, file, null);
    }

    public static String uploadFile(String url, File file, List<Param> headers) throws IOException{
        return uploadFile(url, file, headers, null);
    }

    /**
     * 上传单个文件
     * @param url 服务器地址
     * @param file 上传的文件
     * @param headers 请求头
     * @param params 请求参数
     * */
    public static String uploadFile(String url, File file, List<Param> headers, List<Param> params) throws IOException{
        String result = "";
        Response response = null;
        try {
            response = getInstance().upload(url, file, headers, params);
            if(response.isSuccessful()){
                result = response.body().string();
            }
        } catch (IOException e) {
            throw e;
        }
        return result;
    }

    public static Response uploadFileReturnResonse(String url, File file, List<Param> headers, List<Param> params) throws IOException{
        return getInstance().upload(url, file, headers, params);
    }

    public static String downLoadFile(String url, File saveFile) throws IOException{
        return downLoadFile(url, saveFile, null, null);
    }

    /**
     * 以Get请求的方式请网络上下载文件,并返回保存的文件路径，失败返回空
     * @param url 要下载的文件的url
     * @param saveFile 要保存成的文件
     * @param headers 请求头
     * @param params 请求参数
     * @return 返回文件的绝对路径，有异常返回空
     */
    public static String downLoadFile(String url, File saveFile, List<Param> headers, List<Param> params) throws IOException{
        String saveFilePath = "";
        Response response = null;
        try {
            response = getInstance().downLoad(url, headers, params);
            if(response.isSuccessful()){
                FileCopyUtil.copy(response.body().byteStream(), saveFile);
                long localPicSize = saveFile.length();
                if(localPicSize>0){
                    saveFilePath=saveFile.getAbsolutePath();
                }else{
                    saveFilePath = "";
                }
            }
        } catch (IOException e) {
            throw e;
        }
        return saveFilePath;
    }

    public void cancleCallWithTag(Object tag){
        if(tag == null){
            return;
        }
        synchronized (mOkHttpClient.dispatcher().getClass()){
            for(Call call: mOkHttpClient.dispatcher().queuedCalls()){
                if(tag.equals(call.request().tag())){
                    call.cancel();
                }
            }
        }
    }

    private Response getRequest(String url, List<Param> params) throws  IOException{
        String queryParameter = "";
        if(params != null){
            queryParameter = getRequestParameters(params);
        }
        Uri uri = Uri.parse(url);
        if(TextUtils.isEmpty(uri.getQuery())){
            if(!TextUtils.isEmpty(queryParameter)){
                url = url + "?" + queryParameter;
            }
        }else{
            if(!TextUtils.isEmpty(queryParameter)){
                url = url + "&" + queryParameter;
            }
        }
        final Request getRequest = new Request.Builder().url(url).build();
        return mOkHttpClient.newCall(getRequest).execute();
    }

    private Response postRequest(String url, List<Param> params) throws IOException{
        FormBody.Builder fbBuilder = new FormBody.Builder();
        if(params != null && !params.isEmpty()){
            for(Param param : params){
                fbBuilder.add(param.key, param.value);
            }
        }
        final Request postRequest = new Request.Builder().url(url).post(fbBuilder.build()).build();
        return mOkHttpClient.newCall(postRequest).execute();
    }

    private Response postRequest(String url, String json) throws IOException{
        RequestBody requestBody = RequestBody.create(MediaType.parse("application/json; charset=utf-8"), json);
        final Request postRequest = new Request.Builder().url(url).post(requestBody).build();
        return mOkHttpClient.newCall(postRequest).execute();
    }

    private Response downLoad(String url, List<Param> headers, List<Param> params) throws IOException{
        if(params != null && !params.isEmpty()){
            String parameters = getRequestParameters(params);
            Uri uri = Uri.parse(url);
            if(TextUtils.isEmpty(uri.getQuery())){
                url = url + "?" +parameters;
            }else{
                url = url + "&" +parameters;
            }
        }
        Headers header = null;
        if(headers != null && !headers.isEmpty()){
            header = getRequestHeaders(headers);
        }
        Request.Builder builder = new Request.Builder();
        builder.url(url);
        if(header != null){
            builder.headers(header);
        }
        return mOkHttpClient.newCall(builder.build()).execute();
    }

    private Response upload(String url, File file, List<Param> headers, List<Param> params) throws IOException{
        MultipartBody.Builder multipartBuilder = new MultipartBody.Builder();
        if(headers != null && !headers.isEmpty()){
            multipartBuilder.addPart(getRequestHeaders(headers),RequestBody.create(MediaType.parse(getFileMimeType(file.getName())), file));
        }
        if(params != null && !params.isEmpty()){
            for(Param param : params){
                multipartBuilder.addFormDataPart(param.key, param.value);
            }
        }
        final Request uploadRequest = new Request.Builder().url(url).post(multipartBuilder.build()).build();
        return mOkHttpClient.newCall(uploadRequest).execute();
    }

    private static String getFileMimeType(String fileName){
        FileNameMap fileNameMap = URLConnection.getFileNameMap();
        String contentType = fileNameMap.getContentTypeFor(fileName);
        if(contentType == null){
            contentType = "application/octet-stream";
        }
        return contentType;
    }

    private static Headers getRequestHeaders(List<Param> headers){
        Headers.Builder headerBuilder = null;
        if(headers != null){
            headerBuilder = new Headers.Builder();
            for(Param header : headers){
                headerBuilder.add(header.key, header.value);
            }
        }
        return headerBuilder.build();
    }

    private static String getRequestParameters(List<Param> params){
        StringBuilder paramSb = new StringBuilder();
        for(Param param : params){
            paramSb.append(param.key).append("=").append(param.value).append("&");
        }
        return paramSb.substring(0, paramSb.length()-1);
    }

    private static String getRequestUrl(String url,Map<String, String> params){
        StringBuilder urlSb = new StringBuilder(url);
        urlSb.append('?');
        for(String key : params.keySet()){
            urlSb.append(key).append('=').append(params.get(key)).append("&");
        }
        return urlSb.substring(0, urlSb.length()-1);
    }

    /**
     * post请求参数
     * */
    public static class Param{
        String key;
        String value;

        public Param(){

        }

        public Param(String key, String value){
            this.key = key;
            this.value = value;
        }
    }
}

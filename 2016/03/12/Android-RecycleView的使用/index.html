Android RecycleView的使用

  • View
  • android
学习

RecyclerView控件和ListView的原理有很多相似的地方,都是维护少量的View来进行显示大量的数据，不过RecyclerView控件比ListView更加高级并且更加灵活。RecyclerView不用再负责Item的显示相关的功能，有关布局,绘制,数据绑定等都被分拆成不同的类进行管理。下面这个实例基本覆盖了RecycleView的功能。

/**
 * RecyclerView控件和ListView的原理有很多相似的地方,都是维护少量的View来进行显示大量的数据，不过RecyclerView控件比ListView更加高级并且更加灵活。
 * RecyclerView不用在负责Item的显示相关的功能，在这边所有有关布局,绘制,数据绑定等都被分拆成不同的类进行管理
 * 1，RecycleView.Adapter用来将数据和布局item进行绑定
 * 2，RecycleView.LayoutManager负责item视图的布局和显示管理，分为三种：
 *  2.1 LinearLayoutManager线性布局，横向或纵向滑动列表
 *  2.2 GridLayoutManager表格布局
 *  2.3 StaggeredGridLayoutManager流式布局，例如瀑布流效果
 * 3，RecycleView.ViewHolder承载视图的子布局
 * 4，RecycleView.ItemDecoration给每个item视图添加子view，例如可以进行画分割线
 * 5，RecycleView.ItemAnimator负责处理数据添加或删除时的动画
 *
 * Created by wh on 2016/3/12.
 */
public class RecycleViewTestActivity extends AppCompatActivity{

    private Context mContext = this;
    private RecyclerView mRecycleView;
    private RecycleViewTestAdapter mAdapter;
    private LinearLayoutManager mLayoutManager;

    private List<String> mListData;

    @Override
    protected void onCreate(@Nullable Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.recycle_view_test_activity);

        initView();
        initData();
    }

    private void initView(){
        mRecycleView = (RecyclerView) findViewById(R.id.recycle_view);
        //设置固定大小，可提升性能
        mRecycleView.setHasFixedSize(true);
        //创建线性布局(另外还有GridLayoutManager和StaggeredGridLayoutManager)
        mLayoutManager = new LinearLayoutManager(mContext);
        //设置方向
        mLayoutManager.setOrientation(OrientationHelper.VERTICAL);
        //设置布局
        mRecycleView.setLayoutManager(mLayoutManager);
        //创建适配器
        mAdapter = new RecycleViewTestAdapter();
        //设置item点击事件
        mAdapter.setOnItemClickListener(new RecycleViewTestAdapter.OnItemClickListener() {
            @Override
            public void onItemClick(View v, int position) {
                Toast.makeText(mContext, "Item"+position, Toast.LENGTH_SHORT).show();
            }

            @Override
            public void onItemLongClick(View v, int position) {
                Toast.makeText(mContext, "Item"+position, Toast.LENGTH_SHORT).show();
                //长按删除，有动画效果
                mAdapter.remove(position);
            }
        });
        //设置适配器
        mRecycleView.setAdapter(mAdapter);
        //设置item动画（也可是使用自定义动画，例如：https://github.com/gabrielemariotti/RecyclerViewItemAnimators）
        mRecycleView.setItemAnimator(new DefaultItemAnimator());
        //设置分割线
        mRecycleView.addItemDecoration(new RecycleViewItemDecoration(mContext, RecycleViewItemDecoration.VERTICAL));
        //设置滚动监听
        mRecycleView.addOnScrollListener(new RecyclerView.OnScrollListener() {
            @Override
            public void onScrollStateChanged(RecyclerView recyclerView, int newState) {
                super.onScrollStateChanged(recyclerView, newState);
                int firstVisiblePosition = mLayoutManager.findFirstVisibleItemPosition();
                int lastVisiblePosition = mLayoutManager.findLastVisibleItemPosition();
                int totalItemCount = mLayoutManager.getItemCount();
                if(lastVisiblePosition + 1 >= totalItemCount){
                    loadMore();
                }
            }

            @Override
            public void onScrolled(RecyclerView recyclerView, int dx, int dy) {
                super.onScrolled(recyclerView, dx, dy);
            }
        });
    }

    //列表item数据
    private void initData(){
        mListData = new ArrayList<>();
        for(int i=0;i<16;i++){
            mListData.add("Item" + i);
        }
        mAdapter.addAll(mListData);
    }

    //加载下一页
    private void loadMore(){
        List<String> addList = new ArrayList<>(16);
        int size = mAdapter.getItemCount();
        for(int i=size;i<size+16;i++){
            addList.add("Item"+i);
        }
        mAdapter.addAll(addList);
    }
}

public class RecycleViewTestAdapter extends SimpleAdapter<String,RecycleViewTestAdapter.ViewHolder>{
    private OnItemClickListener mOnItemClickListener;

    public interface OnItemClickListener{
        void onItemClick(View v, int position);
        void onItemLongClick(View v, int position);
    }

    public void setOnItemClickListener(OnItemClickListener onItemClickListener){
        this.mOnItemClickListener = onItemClickListener;
    }

    @Override
    public ViewHolder onCreateViewHolder(ViewGroup parent, int viewType) {
        View view = LayoutInflater.from(parent.getContext()).inflate(R.layout.recycle_view_item_layout, parent, false);
        ViewHolder viewHolder = new ViewHolder(view);
        /*view.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                if(mOnItemClickListener != null){
                    mOnItemClickListener.onItemClick(v, (int) v.getTag());
                }
            }
        });*/
        return viewHolder;
    }

    @Override
    public void onBindViewHolder(final ViewHolder holder, int position) {
        holder.mItemTv.setText(getItem(position));
        //holder.itemView.setTag(position);//设置一个tag,用于点击事件
        //如果设置了点击回掉，则设置点击事件
        if(mOnItemClickListener != null){
            holder.itemView.setOnClickListener(new View.OnClickListener() {
                @Override
                public void onClick(View v) {
                    int position = holder.getLayoutPosition();
                    mOnItemClickListener.onItemClick(holder.itemView, position);
                }
            });

            holder.itemView.setOnLongClickListener(new View.OnLongClickListener() {
                @Override
                public boolean onLongClick(View v) {
                    int position = holder.getLayoutPosition();
                    mOnItemClickListener.onItemLongClick(holder.itemView, position);
                    return false;
                }
            });
        }
    }


    public static class ViewHolder extends RecyclerView.ViewHolder{
        public TextView mItemTv;
        public ViewHolder(View itemView) {
            super(itemView);
            mItemTv = (TextView) itemView.findViewById(R.id.item_tv);
        }
    }
}

    public static boolean createFile(String fileName){
        boolean result = false;
        File newFile = new File(dirPath + fileName);
        try {
            result = newFile.createNewFile();
        } catch (IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
        return result;
    }

    public static boolean deleteFile(String fileName){
        boolean result = false;
        File file = new File(dirPath + fileName);
        if(file.exists()){
            result = file.delete();
        }
        return result;
    }

    public static void listAllFileNames(String path){
        File file = new File(path);
        String[] fileNames = file.list();
        if(fileNames != null){
            for(int i=0;i<fileNames.length;i++){
                System.out.println(fileNames[i]);
            }
        }
    }

    public static void listFiles(String path){
        File file = new File(path);
        File[] files = file.listFiles();
        if(files != null){
            for(int i=0;i<files.length;i++){
                System.out.println(files[i]);
            }
        }
    }

    public static boolean isDir(String path){
        boolean result = false;
        File file = new File(path);
        if(file.exists()){
            System.out.println(file.isDirectory());
            result = file.isDirectory();
        }else{
            System.out.println("file is not exists");
            result = false;
        }
        return result;
    }

    public static void listAllFiles(String path){
        File file = new File(path);
        list(file);
    }

    private static void list(File f){
        if(f != null){
            if(f.isDirectory()){
                File[] files = f.listFiles();
                if(files != null){
                    for(int i=0;i<files.length;i++){
                        list(files[i]);
                    }
                }
            }else{
                System.out.println(f);
            }
        }
    }

    public static void outputWithRAF(String fileName){
        File file = new File(dirPath+fileName);
        try {
            RandomAccessFile raf = new RandomAccessFile(file, "rw");
            raf.writeBytes("hello world");
            raf.writeInt(123);
            raf.writeBoolean(true);
            raf.writeChar('W');
        } catch (FileNotFoundException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        } catch (IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }

    }

    //OutputStream and InputStream
    public static void copyFile(String from,String to){
        File srcFile = new File(from);
        File desFile = new File(to);
        OutputStream os=null;
        InputStream is=null;
        try {
            os = new FileOutputStream(desFile);
            is = new FileInputStream(srcFile);
            byte[] buffer = new byte[1024];
            int i;
            while((i=is.read(buffer)) != -1){
                os.write(buffer, 0, i);
            }
        } catch (FileNotFoundException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        } catch (IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        } finally{
            try {
                is.close();
                os.close();
            } catch (IOException e) {
                // TODO Auto-generated catch block
                e.printStackTrace();
            }
        }

    }

    //Writer AND Reader
    public static void writerReader(String path){
        File file = new File(path);
        BufferedWriter writer = null;
        BufferedReader reader = null;
        try {
            writer = new BufferedWriter(new FileWriter(file));
            writer.write("hello world");
            writer.flush();
            writer.close();

            reader = new BufferedReader(new FileReader(file));
            String line = null;
            while((line = reader.readLine()) != null){
                System.out.println(line);
            }
            reader.close();
        } catch (IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        } 
    }

    //字节字符转换流
    public static void streamChange(String path){
        File file = new File(dirPath+path);
        Writer out = null;
        Reader in = null;
        try {
            out = new OutputStreamWriter(new FileOutputStream(file));
            out.write("hello world");
            out.close();

            in = new InputStreamReader(new FileInputStream(file));
            char[] buffer = new char[100];
            int c = in.read(buffer);
            in.close();
            System.out.println(new String(buffer,0,c));

        } catch (FileNotFoundException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        } catch (IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
    }

    //对象序列化
    public static void objectSerialize(String path){
        File file = new File(dirPath+path);
        ObjectOutputStream oos;
        try {
            oos = new ObjectOutputStream(new FileOutputStream(file));
            oos.writeObject(new Person("java", 22));
            oos.close();
        } catch (IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
    }

    //对象反序列化
    public static void objectSerializeReverse(String path){
        File file = new File(dirPath+path);
        ObjectInputStream ois;
        try {
            ois = new ObjectInputStream(new FileInputStream(file));
            Person person = (Person)ois.readObject();
            ois.close();
            System.out.println(person);
        } catch (IOException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        } catch (ClassNotFoundException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
    }
}


class Person implements Serializable{
    private String name;
    private int age;

    public Person(){

    }

    public Person(String name,int age){
        this.name = name;
        this.age = age;
    }

    @Override
    public String toString() {
        return "姓名:"+name+"；年龄:"+age;
    }
}
